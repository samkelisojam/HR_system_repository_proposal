Employee Management Microservice

Employee Management Microservice

Responsibilities

Onboard new employees, create and manage employee profiles, assign company ID, department, reporting manager, and roles, and update employee details as needed.

Endpoints

  • POST /employees - Create a new employee.
  • GET /employees/{id} - Retrieve employee details.
  • PUT /employees/{id} - Update employee details.
  • DELETE /employees/{id} - Delete an employee profile.

Database Models

Employee Table

Field Name Data Type Constraints
employee_id INT PRIMARY KEY, AUTO_INCREMENT
first_name VARCHAR(50) NOT NULL
last_name VARCHAR(50) NOT NULL
department_id INT FOREIGN KEY (references Department)
reporting_manager INT FOREIGN KEY (references Employee)
role_id INT FOREIGN KEY (references Role)
company_id VARCHAR(20) UNIQUE, NOT NULL
email VARCHAR(100) UNIQUE, NOT NULL
phone_number VARCHAR(15) UNIQUE
hire_date DATE NOT NULL

Essential Methods and Helpers

public class EmployeeManagementService
{
    private readonly EmployeeRepository _employeeRepository;
    private readonly DepartmentRepository _departmentRepository;
    private readonly RoleRepository _roleRepository;

    public EmployeeManagementService(EmployeeRepository employeeRepository, DepartmentRepository departmentRepository, RoleRepository roleRepository)
    {
        _employeeRepository = employeeRepository;
        _departmentRepository = departmentRepository;
        _roleRepository = roleRepository;
    }

    public Employee CreateEmployee(Employee employee)
    {
        // Validate employee data
        // Assign company ID, department, reporting manager, and roles
        // Save employee to the database
        return _employeeRepository.Add(employee);
    }

    public Employee GetEmployeeById(int id)
    {
        return _employeeRepository.GetById(id);
    }

    public Employee UpdateEmployee(int id, Employee updatedEmployee)
    {
        // Validate updated employee data
        // Update employee details in the database
        return _employeeRepository.Update(id, updatedEmployee);
    }

    public void DeleteEmployee(int id)
    {
        // Remove employee from the database
        _employeeRepository.Delete(id);
    }
}
